import cv2
import socket
import requests
import numpy as np

# ================= BODY DETECTOR CLASS =================
class BodyDetector:
    def __init__(self):
        self.cascades = [
            ("fullbody", cv2.CascadeClassifier(cv2.data.haarcascades + "haarcascade_fullbody.xml")),
            ("upperbody", cv2.CascadeClassifier(cv2.data.haarcascades + "haarcascade_upperbody.xml")),
            ("lowerbody", cv2.CascadeClassifier(cv2.data.haarcascades + "haarcascade_lowerbody.xml")),
            ("profile", cv2.CascadeClassifier(cv2.data.haarcascades + "haarcascade_profileface.xml"))
        ]
        self.active_cascade = self.cascades[0]

    def detect_largest_body(self, gray):
        best_rect = None
        max_area = 0

        for cascade_name, cascade in self.cascades:
            if cascade.empty():
                continue

            bodies = cascade.detectMultiScale(gray, 1.1, 3, minSize=(60, 60))

            for (x, y, w, h) in bodies:
                area = w * h
                if area > max_area:
                    max_area = area
                    best_rect = (x, y, w, h)
                    self.active_cascade = (cascade_name, cascade)

        return best_rect


# ================= COMMUNICATION CLASS =================
class RobotClient:
    def __init__(self, pi_ip, pi_port):
        self.sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        self.sock.connect((pi_ip, pi_port))
        self.sock.settimeout(1.0)

    def send_command(self, cmd):
        self.sock.sendall((cmd + "\n").encode())


# ================= MAIN BODY TRACKER =================
class BodyTracker:
    def __init__(self, ip_cam_url, pi_ip, pi_port):
        self.ip_cam_url = ip_cam_url
        self.detector = BodyDetector()
        self.robot = RobotClient(pi_ip, pi_port)

    def run(self):
        while True:
            frame = cv2.imread("frame.jpg")
            gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
            body = self.detector.detect_largest_body(gray)

            if body:
                self.robot.send_command("FORWARD")
            else:
                self.robot.send_command("STOP")

Raspberry Pi Side (Actuation + Safety)

import RPi.GPIO as GPIO
import time
import socket

# ================= ACTUATOR CLASS =================
class Actuator:
    def __init__(self, left_pin=23, right_pin=24):
        GPIO.setup(left_pin, GPIO.OUT)
        GPIO.setup(right_pin, GPIO.OUT)

        self.left_pwm = GPIO.PWM(left_pin, 50)
        self.right_pwm = GPIO.PWM(right_pin, 50)

        self.left_pwm.start(0)
        self.right_pwm.start(0)

    def forward(self):
        self.left_pwm.ChangeDutyCycle(8)
        self.right_pwm.ChangeDutyCycle(8)

    def stop(self):
        self.left_pwm.ChangeDutyCycle(7.5)
        self.right_pwm.ChangeDutyCycle(7.5)


# ================= CONTROLLER CLASS =================
class RobotController:
    def __init__(self):
        GPIO.setmode(GPIO.BCM)
        self.actuator = Actuator()

        self.server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        self.server.bind(("0.0.0.0", 9999))
        self.server.listen(1)
        self.conn, _ = self.server.accept()

    def run(self):
        while True:
            cmd = self.conn.recv(1024).decode().strip()
            if cmd == "FORWARD":
                self.actuator.forward()
            else:
                self.actuator.stop()
